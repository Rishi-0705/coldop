    41→export default function ColdOpsPage() {
    42→  const [view, setView] = useState<ViewKey>('command')
    43→  const [dashboard, setDashboard] = useState<DashboardData | null>(null)
    44→  const [rooms, setRooms] = useState<RoomWithBms[]>([])
    45→  const [notifs, setNotifs] = useState<Notification[]>([])
    46→  const [workOrders, setWorkOrders] = useState<WorkOrder[]>([])
    47→  const [plan, setPlan] = useState<ConsolidationPlan | null>(null)
    48→  const [activeSetbacks, setActiveSetbacks] = useState<ActiveSetback[]>([])
    49→  const [analytics, setAnalytics] = useState<AnalyticsData | null>(null)
    50→  const [meterData, setMeterData] = useState<MeterData | null>(null)
    51→  const [setbackHistory, setSetbackHistory] = useState<SetbackHistoryItem[]>([])
    52→  const [schedule, setSchedule] = useState<ProductionScheduleData | null>(null)
    53→  const [loading, setLoading] = useState(true)
    54→  const [notifCounts, setNotifCounts] = useState<Record<string, number>>({ CRITICAL: 0, HIGH: 0, MEDIUM: 0, LOW: 0 })
    55→  const [bmsOnline, setBmsOnline] = useState(true)
    56→  const [realtimeConnected, setRealtimeConnected] = useState(false)
    57→  const [tick, setTick] = useState(0)
    58→
    59→  const fetchAll = useCallback(async () => {
    60→    try {
    61→      const [d, r, n, w, c, s, bh] = await Promise.all([
    62→        fetch('/api/dashboard').then(r => r.json()),
    63→        fetch('/api/rooms').then(r => r.json()),
    64→        fetch('/api/notifications').then(r => r.json()),
    65→        fetch('/api/work-orders').then(r => r.json()),
    66→        fetch('/api/consolidation').then(r => r.json()),
    67→        fetch('/api/setback/active').then(r => r.json()),
    68→        fetch('/api/bms-health').then(r => r.json()),
    69→      ])
    70→      setDashboard(d)
    71→      setRooms(r.rooms || [])
    72→      setNotifs(n.notifications || [])
    73→      setNotifCounts(n.counts || {})
    74→      setWorkOrders(w.workOrders || [])
    75→      setPlan(c.plan)
    76→      setActiveSetbacks(s.active || [])
    77→      setBmsOnline(bh.online)
    78→    } catch (e) {
    79→      console.error('fetch failed', e)
    80→    } finally {
    81→      setLoading(false)
    82→    }
    83→  }, [])
    84→
    85→  // Fetch analytics data when analytics view is opened
    86→  const fetchAnalytics = useCallback(async () => {
    87→    try {
    88→      const [a, m, sh] = await Promise.all([
    89→        fetch('/api/analytics').then(r => r.json()),
    90→        fetch('/api/meter-readings?hours=6').then(r => r.json()),
    91→        fetch('/api/setbacks').then(r => r.json()),
    92→      ])
    93→      setAnalytics(a)
    94→      setMeterData(m)
    95→      setSetbackHistory(sh.setbacks || [])
    96→    } catch (e) {
    97→      console.error('analytics fetch failed', e)
    98→    }
    99→  }, [])
   100→
   101→  // Fetch meter data for the Command Center chart
   102→  const fetchMeter = useCallback(async () => {
   103→    try {
   104→      const m = await fetch('/api/meter-readings?hours=6').then(r => r.json())
   105→      setMeterData(m)
   106→    } catch (e) {
   107→      console.error('meter fetch failed', e)
   108→    }
   109→  }, [])
   110→
   111→  // Fetch schedule data when schedule view is opened
   112→  const fetchSchedule = useCallback(async () => {
   113→    try {
   114→      const s = await fetch('/api/production-schedule').then(r => r.json())
   115→      setSchedule(s)
   116→    } catch (e) {
   117→      console.error('schedule fetch failed', e)
   118→    }
   119→  }, [])
   120→
   121→  useEffect(() => {
   122→    if (view === 'analytics' && !analytics) fetchAnalytics()
   123→    if (view === 'schedule' && !schedule) fetchSchedule()
   124→  }, [view, analytics, schedule, fetchAnalytics, fetchSchedule])
   125→
   126→  useEffect(() => {
   127→    fetchAll()
   128→    fetchMeter()
   129→    const interval = setInterval(() => setTick(t => t + 1), 8000)
   130→    return () => clearInterval(interval)
   131→  }, [fetchAll, fetchMeter])
   132→
   133→  useEffect(() => {
   134→    if (tick > 0) fetchAll()
   135→  }, [tick, fetchAll])
   136→
   137→  // Real-time events
   138→  const conn = useRealtimeConnection()
   139→  useEffect(() => setRealtimeConnected(conn), [conn])
   140→
   141→  useRealtimeEvent('savings-updated', (p: Partial<Savings>) => {
   142→    setDashboard(d => d ? { ...d, savings: { ...d.savings, ...p } as Savings } : d)
   143→  })
   144→  useRealtimeEvent('notification-new', (n: Notification) => {
   145→    setNotifs(prev => [n, ...prev])
   146→    setNotifCounts(prev => ({ ...prev, [n.severity]: (prev[n.severity] || 0) + 1 }))
   147→    toast.info(`New ${n.severity.toLowerCase()} alert: ${n.title}`)
   148→  })
   149→  useRealtimeEvent('notification-updated', (p: { id: string; status: string }) => {
   150→    setNotifs(prev => prev.map(n => n.id === p.id ? { ...n, status: p.status as Notification['status'] } : n))
   151→    fetchAll()
   152→  })
   153→  useRealtimeEvent('setback-progress', () => {
   154→    fetch('/api/setback/active').then(r => r.json()).then(s => setActiveSetbacks(s.active || []))
   155→  })
   156→  useRealtimeEvent('setback-completed', (p: { setbackId: string }) => {
   157→    setActiveSetbacks(prev => prev.filter(s => s.setbackId !== p.setbackId))
   158→    toast.success(`Setback completed — savings accumulating`)
   159→    fetchAll()
   160→  })
   161→  useRealtimeEvent('setback-aborted', (p: { setbackId: string; reason: string }) => {
   162→    setActiveSetbacks(prev => prev.filter(s => s.setbackId !== p.setbackId))
   163→    toast.error(`Setback aborted: ${p.reason}`)
   164→    fetchAll()
   165→  })
   166→  useRealtimeEvent('work-order-completed', () => {
   167→    fetchAll()
   168→    toast.success('Work order completed')
   169→  })
   170→  useRealtimeEvent('work-order-updated', () => fetchAll())
   171→  useRealtimeEvent('ghost-load-detected', () => fetchAll())
   172→  useRealtimeEvent('room-status-changed', () => fetchAll())
   173→
   174→  return (
   175→    <div className="min-h-screen flex flex-col bg-background">
   176→      <TopBar
   177→        view={view}
   178→        onView={setView}
   179→        notifCounts={notifCounts}
   180→        bmsOnline={bmsOnline}
   181→        realtimeConnected={realtimeConnected}
   182→        savings={dashboard?.savings}
   183→        onRefresh={() => { setLoading(true); fetchAll() }}
   184→      />
   185→      <main className="flex-1 container mx-auto px-4 py-6 max-w-[1600px]">
   186→        {loading && !dashboard ? (
   187→          <LoadingState />
   188→        ) : view === 'command' ? (
   189→          <CommandCenter dashboard={dashboard} rooms={rooms} activeSetbacks={activeSetbacks} meterData={meterData} onNeedMeter={fetchMeter} />
   190→        ) : view === 'map' ? (
   191→          <ColdRoomMap rooms={rooms} plan={plan} onExecutePlan={fetchAll} />
   192→        ) : view === 'workorders' ? (
   193→          <WorkOrdersView workOrders={workOrders} activeSetbacks={activeSetbacks} onComplete={fetchAll} />
   194→        ) : view === 'notifications' ? (
   195→          <NotificationsView notifs={notifs} counts={notifCounts} onAction={fetchAll} />
   196→        ) : view === 'analytics' ? (
   197→          <AnalyticsView analytics={analytics} meterData={meterData} setbackHistory={setbackHistory} onRefresh={fetchAnalytics} />
   198→        ) : (
   199→          <ScheduleView schedule={schedule} onRefresh={fetchSchedule} />
   200→        )}
   201→      </main>
   202→      <Footer />
   203→    </div>
   204→  )
   205→}
