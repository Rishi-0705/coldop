    72→    72→      setNotifs(n.notifications || [])

   108→   108→    }

   163→   163→    toast.error(`Setback aborted: ${p.reason}`)
   164→   164→    fetchAll()
   165→   165→  })
   166→   166→  useRealtimeEvent('work-order-completed', () => {
   167→   167→    fetchAll()
   168→   168→    toast.success('Work order completed')
   169→   169→  })
   170→   170→  useRealtimeEvent('work-order-updated', () => fetchAll())
   171→   171→  useRealtimeEvent('ghost-load-detected', () => fetchAll())
   172→   172→  useRealtimeEvent('room-status-changed', () => fetchAll())
   173→   173→
   174→   174→  return (
   175→   175→    <div className="min-h-screen flex flex-col bg-background">
   176→   176→      <TopBar

   184→   184→      />

   186→   186→        {loading && !dashboard ? (

   190→   190→        ) : view === 'map' ? (
   191→   191→          <ColdRoomMap rooms={rooms} plan={plan} onExecutePlan={fetchAll} />

   193→   193→          <WorkOrdersView workOrders={workOrders} activeSetbacks={activeSetbacks} onComplete={fetchAll} />
   194→   194→        ) : view === 'notifications' ? (

   198→   198→        ) : (
   199→   199→          <ScheduleView schedule={schedule} onRefresh={fetchSchedule} />
